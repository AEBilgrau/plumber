Decorating

#* @get /salesgraph
#* @png
function(){
  par(mar = rep(2, 4)) # margins
  dates <- seq(as.Date("2015-10-01"),
      as.Date("2015-10-31"), by=1)
  sold <- 1:31

  plot(dates, sold, type="b", main="Sales")
}

Decorating

sales <- NULL

#* @post /addsale
function(qty){
  sales <- rbind(sales, list(
      time = Sys.time(),
      item="hat",
      qty = qty
  ))
}
